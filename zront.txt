DATABASE SCHEMA — REAL TABLES:

Table: customers
  - id: uuid, primary key
  - first_name: varchar(100), not null
  - last_name: varchar(100), not null
  - email: varchar(255), unique, not null
  - phone: varchar(20)
  - created_at: timestamp, default now()
  - updated_at: timestamp

Table: addresses
  - id: uuid, primary key
  - customer_id: uuid, foreign key → customers.id
  - address_line1: varchar(255), not null
  - address_line2: varchar(255)
  - city: varchar(100), not null
  - state: varchar(100), not null
  - country: varchar(100), not null, default 'Nigeria'
  - is_default: boolean, default false
  - created_at: timestamp, default now()

Table: products
  - id: uuid, primary key
  - name: varchar(255), not null
  - slug: varchar(255), unique, not null
  - description: text
  - category: enum('hair_care','men','women',
    'bags','shoes'), not null
  - price: numeric(10,2), not null
  - compare_at_price: numeric(10,2)
    (for showing original price if on sale)
  - stock_quantity: integer, not null, default 0
  - is_available: boolean, default true
  - images: jsonb 
    (array of image URLs with primary flag)
  - benefits: jsonb
    (array of benefit objects {icon, label})
  - created_at: timestamp, default now()
  - updated_at: timestamp

Table: orders
  - id: uuid, primary key
  - order_number: varchar(20), unique, not null
    (human readable e.g. LUX-2026-00142)
  - customer_id: uuid, foreign key → customers.id
  - shipping_address_id: uuid, 
    foreign key → addresses.id
  - status: enum('pending', 'confirmed', 
    'processing', 'shipped', 'delivered', 
    'cancelled'), default 'pending'
  - subtotal: numeric(10,2), not null
  - shipping_fee: numeric(10,2), not null
  - total: numeric(10,2), not null
  - notes: text (customer order notes)
  - created_at: timestamp, default now()
  - updated_at: timestamp

Table: order_items
  - id: uuid, primary key
  - order_id: uuid, foreign key → orders.id
  - product_id: uuid, foreign key → products.id
  - product_name: varchar(255), not null
    (snapshot of name at time of order)
  - product_price: numeric(10,2), not null
    (snapshot of price at time of order)
  - quantity: integer, not null
  - subtotal: numeric(10,2), not null
  - created_at: timestamp, default now()

Table: order_status_history
  - id: uuid, primary key
  - order_id: uuid, foreign key → orders.id
  - status: enum (same as orders.status)
  - note: text (internal admin note per status)
  - changed_at: timestamp, default now()

API CALLS:

POST /api/orders
  Request body:
  {
    customer: {
      first_name, last_name, email, phone
    },
    shipping_address: {
      address_line1, address_line2,
      city, state, country
    },
    items: [
      { product_id, quantity }
    ],
    notes: string (optional)
  }
  Response:
  {
    order_id, order_number, status, total,
    created_at
  }

GET /api/orders/:order_number
  Returns full order with items and status history
